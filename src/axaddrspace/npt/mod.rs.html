28pub enum NestedPageTable<H: PagingHandler> {
29    #[cfg(not(target_arch = "x86_64"))]
30    L3(NestedPageTableL3<H>),
31    L4(NestedPageTableL4<H>),
32}
33
34impl<H: PagingHandler> NestedPageTable<H> {
35    pub fn new(level: usize) -> axerrno::AxResult<Self> {
36        match level {
37            3 => {
38                #[cfg(not(target_arch = "x86_64"))]
39                {
40                    use axerrno::ax_err_type;
41
42                    let res = NestedPageTableL3::try_new().map_err(|_| ax_err_type!(NoMemory))?;
43                    return Ok(NestedPageTable::L3(res));
44                }
45                #[cfg(target_arch = "x86_64")]
46                {
47                    return ax_err!(InvalidInput, "L3 not supported on x86_64");
48                }
49            }
50            4 => {
51                let res = NestedPageTableL4::try_new().map_err(|_| ax_err_type!(NoMemory))?;
52                return Ok(NestedPageTable::L4(res));
53            }
54            _ => return ax_err!(InvalidInput, "Invalid page table level"),
55        }
56    }
57
58    pub fn root_paddr(&self) -> memory_addr::PhysAddr {
59        match self {
60            #[cfg(not(target_arch = "x86_64"))]
61            NestedPageTable::L3(pt) => pt.root_paddr(),
62            NestedPageTable::L4(pt) => pt.root_paddr(),
63        }
64    }
65
66    /// Maps a virtual address to a physical address.
67    pub fn map(
68        &mut self,
69        vaddr: crate::GuestPhysAddr,
70        paddr: memory_addr::PhysAddr,
71        size: page_table_multiarch::PageSize,
72        flags: page_table_entry::MappingFlags,
73    ) -> memory_set::MappingResult {
74        match self {
75            #[cfg(not(target_arch = "x86_64"))]
76            NestedPageTable::L3(pt) => {
77                pt.map(vaddr, paddr, size, flags)
78                    .map_err(|_| MappingError::BadState)?
79                    .flush();
80            }
81            NestedPageTable::L4(pt) => {
82                let _res = pt
83                    .map(vaddr, paddr, size, flags)
84                    .map_err(|_| MappingError::BadState)?
85                    .flush();
86            }
87        }
88        Ok(())
89    }
90
91    /// Unmaps a virtual address.
92    pub fn unmap(
93        &mut self,
94        vaddr: GuestPhysAddr,
95    ) -> memory_set::MappingResult<(memory_addr::PhysAddr, page_table_multiarch::PageSize)> {
96        match self {
97            #[cfg(not(target_arch = "x86_64"))]
98            NestedPageTable::L3(pt) => {
99                let (addr, size, f) = pt.unmap(vaddr).map_err(|_| MappingError::BadState)?;
100                f.flush();
101                Ok((addr, size))
102            }
103            NestedPageTable::L4(pt) => {
104                let (addr, size, f) = pt.unmap(vaddr).map_err(|_| MappingError::BadState)?;
105                f.flush();
106                Ok((addr, size))
107            }
108        }
109    }
110
111    /// Maps a region.
112    pub fn map_region(
113        &mut self,
114        vaddr: GuestPhysAddr,
115        get_paddr: impl Fn(GuestPhysAddr) -> PhysAddr,
116        size: usize,
117        flags: MappingFlags,
118        allow_huge: bool,
119        flush_tlb_by_page: bool,
120    ) -> memory_set::MappingResult {
121        match self {
122            #[cfg(not(target_arch = "x86_64"))]
123            NestedPageTable::L3(pt) => {
124                pt.map_region(vaddr, get_paddr, size, flags, allow_huge, flush_tlb_by_page)
125                    .map_err(|_| MappingError::BadState)?
126                    .flush_all();
127            }
128            NestedPageTable::L4(pt) => {
129                pt.map_region(vaddr, get_paddr, size, flags, allow_huge, flush_tlb_by_page)
130                    .map_err(|_| MappingError::BadState)?
131                    .flush_all();
132            }
133        }
134        Ok(())
135    }
136
137    /// Unmaps a region.
138    pub fn unmap_region(
139        &mut self,
140        start: GuestPhysAddr,
141        size: usize,
142        flush: bool,
143    ) -> memory_set::MappingResult {
144        match self {
145            #[cfg(not(target_arch = "x86_64"))]
146            NestedPageTable::L3(pt) => {
147                pt.unmap_region(start, size, flush)
148                    .map_err(|_| MappingError::BadState)?
149                    .ignore();
150            }
151            NestedPageTable::L4(pt) => {
152                pt.unmap_region(start, size, flush)
153                    .map_err(|_| MappingError::BadState)?
154                    .ignore();
155            }
156        }
157        Ok(())
158    }

160    pub fn remap(&mut self, start: GuestPhysAddr, paddr: PhysAddr, flags: MappingFlags) -> bool {
161        match self {
162            #[cfg(not(target_arch = "x86_64"))]
163            NestedPageTable::L3(pt) => pt.remap(start, paddr, flags).is_ok(),
164            NestedPageTable::L4(pt) => pt.remap(start, paddr, flags).is_ok(),
165        }
166    }

168    /// Updates protection flags for a region.
169    pub fn protect_region(
170        &mut self,
171        start: GuestPhysAddr,
172        size: usize,
173        new_flags: page_table_entry::MappingFlags,
174        flush: bool,
175    ) -> bool {
176        match self {
177            #[cfg(not(target_arch = "x86_64"))]
178            NestedPageTable::L3(pt) => pt
179                .protect_region(start, size, new_flags, flush) // If the TLB is refreshed immediately every time, there might be performance issues.
180                // The TLB refresh is managed uniformly at a higher level.
181                .map(|tlb| tlb.ignore())
182                .is_ok(),
183            NestedPageTable::L4(pt) => pt
184                .protect_region(start, size, new_flags, flush) // If the TLB is refreshed immediately every time, there might be performance issues.
185                // The TLB refresh is managed uniformly at a higher level.
186                .map(|tlb| tlb.ignore())
187                .is_ok(),
188        }
189    }

191    /// Queries a virtual address to get physical address and mapping info.
192    pub fn query(
193        &self,
194        vaddr: crate::GuestPhysAddr,
195    ) -> page_table_multiarch::PagingResult<(
196        memory_addr::PhysAddr,
197        page_table_entry::MappingFlags,
198        page_table_multiarch::PageSize,
199    )> {
200        match self {
201            #[cfg(not(target_arch = "x86_64"))]
202            NestedPageTable::L3(pt) => pt.query(vaddr),
203            NestedPageTable::L4(pt) => pt.query(vaddr),
204        }
205    }

207    /// Translates a virtual address to a physical address.
208    pub fn translate(&self, vaddr: crate::GuestPhysAddr) -> Option<crate::HostPhysAddr> {
209        self.query(vaddr).ok().map(|(paddr, _, _)| paddr)
210    }
